120void MqttNowClient::_reconnect() {
121 // Loop until we're reconnected
122 while (!client.connected()) {

164void MqttNowClient::update() {
165 MqttNowBase::update();
166
167 // Store UART input in a local buffer, until
168 // CR (13) or LF (10) is received
169 while (COM.available()) {
170 char c = COM.read();
171 if (c == 10 || c == 13) {
172 if (_handleComm() == result_error) {
173 PRINTLNS("Error handling command!");
174 COM.println(RET_ERROR);
175 } else {
176 COM.println(RET_OK);
177 }
178 _comBuff = "";
179 } else {
180 _comBuff += c;
181 }
182 }
183
184 // Now check if a message was received from the broker
185 if (mqttReceived) {
186 if (lastReceivedTopic.equals(_cmdTopic)) {
187 // A message on the command topic is meant for the client itself
188 _handleCommand();
189 } else {
190 // All other messages are for a node in the ESP-NOW network,
191 // so let the controller handle that
192 _sendMsgToController();
193 }
194 mqttReceived = false;
195 }
196
197 // Handle MQTT stuff
198 if (client.connected()) {
199 client.loop();
200 } else {
201 _reconnect();
202 }
203};

213result_t MqttNowClient::_handleComm() {
214 PRINTLN("Communication received: ", _comBuff);
215
216 if (!_comBuff.startsWith(MSG_START)) {
217 PRINTLNS("Unknown communication");
218 return result_error;
219 }
220
221 // Get character after MSG_START (the action tag)
222 char act = _comBuff.charAt(3);
223
224 switch (act) {
225 case MSG_ACTIONSUB:
226 PRINTLNS("Subscribe command received");
227 return _handleSubscribe();
228 case MSG_ACTIONUNS:
229 PRINTLNS("Unsubscribe command received");
230 return _handleUnsubscribe();
231 case MSG_ACTIONPUB:
232 PRINTLNS("Publish command received");
233 return _handlePublish();
234 default:
235 PRINTLNS("Unknown command received");
236 return result_error;
237 }
238}

240result_t MqttNowClient::_handleSubscribe() {
241 // Everything after action tag is the topic to subscribe to
242 char qos = _comBuff.charAt(4);
243 String topic = _modTopic(_comBuff.substring(5));
244 PRINTF2("Subscribing to topic: %s with QOS %c", topic.c_str(), qos);
245 if (!client.subscribe(topic.c_str(), qos - '0')) {
246 PRINTLNS("Error during subscribing");
247 return result_error;
248 }
249 return result_success;
250}

262result_t MqttNowClient::_handlePublish(String com) {
263 static uint8_t ctr = 0;
264
265 if (com.equals("")) {
266 com = String(_comBuff.c_str());
267 }
268 char qos = com.charAt(4);
269 char retained = com.charAt(5);
270 PRINTLN("QOS = ", qos);
271 PRINTLN("Retained = ", retained);
272
273 int payloadPos = com.lastIndexOf(MSG_PAYLOAD);
274 String topic = _modTopic(com.substring(6,payloadPos));
275 String payload = com.substring(payloadPos + 3);
276
277 PRINTLN("Topic = ", topic);
278 PRINTLN("payload = ", payload);
279
280 // Try max 5 times to publish this
281 if (publish(topic, payload, (retained == '1'), (qos - '0')) == result_success) {
282 ctr = 0;
283 return result_success;
284 } else {
285 if (ctr < 5) {
286 ctr ++;
287 yield();
288 PRINTLN("Publish attempt nr ", (ctr+1));
289 return _handlePublish(com);
290 } else {
291 // Reached max of 5 attempts, return error
292 PRINTLNS("Error publishing message");
293 ctr = 0;
294 return result_error;
295 }
296 }

366 PRINTS("Sending to controller: ");

372

374 size_t send = 0;

377 send += COM.print(lastReceivedTopic);
