116esp_err_t MqttNowNode::addPeer(uint8_t *mac_addr, uint8_t channel, bool encrypt) {
117 #ifdef ESP32
118 esp_now_peer_info_t info;
119 memcpy(info.peer_addr, mac_addr, 6);
120 info.channel = channel;
121 info.encrypt = encrypt;
122 if (encrypt == true) {
123 memcpy(info.lmk, _lmk, ESP_NOW_KEY_LEN);
124 }
125 return addPeer(&info);
126 #elif defined(ESP8266)
127 // ESP 8266 has a simple method with std arguments to add a peer
128 if (encrypt) {
129 return esp_now_add_peer(mac_addr, ESP_NOW_ROLE_SLAVE, channel, _lmk, 16);
130 } else {
131 return esp_now_add_peer(mac_addr, ESP_NOW_ROLE_SLAVE, channel, NULL, 0);
132 }
133 #endif
134}

190result_t MqttNowNode::getMessageStruct(msgType type, msg_base *msg) {
191 switch (type) {
192 case msgTypeIntro:

195 case msgTypeWelcome:
